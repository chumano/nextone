--sync Master.User -> Com.UserStatus
CREATE OR ALTER TRIGGER [master].update_user_status_name ON [master].[User]
AFTER UPDATE
AS
BEGIN
	UPDATE [com].[T_App_UserStatus]
    SET    [com].[T_App_UserStatus].UserName = [INSERTED].Name
    FROM   INSERTED
    WHERE  INSERTED.Id = [com].[T_App_UserStatus].UserId
END


SELECT TOP (1000) us.[UserId]
      ,us.[UserName]
      ,us.[LastUpdateDate]
	  ,u.Name,u.Phone
  FROM [NextOne].[com].[T_App_UserStatus] us
  Join [NextOne].[master].[User] u on us.UserId = u.Id
  where us.UserName <> u.Name

  update us
  set us.UserName = u.Name
  FROM [NextOne].[com].[T_App_UserStatus] us
  Join [NextOne].[master].[User] u on us.UserId = u.Id
  where us.UserName <> u.Name
